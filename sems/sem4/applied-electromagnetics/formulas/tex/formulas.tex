\chapter*{Topic 4 Electrostatics}

\begin{outline}
	\1 $\rho_V = \dfrac{dq}{dV}$
	\1 $\rho_s = \dfrac{dq}{ds}$
	\1 $\rho_l = \dfrac{dq}{dl}$
	\1 General Maxwell equations
	\2 $\nabla\cdot\mathbf{D}  =\rho_{\mathrm{v}}$
	\2 $\mathrm{V \times E}  =-\dfrac{\partial\mathbf{B}}{\partial t}$
	\2 $\nabla\cdot\mathbf{B}  =0$
	\2 $\mathrm{V \times H}  =\mathbf{J}+\dfrac{\partial\mathbf{D}}{\partial t}$
	\1 Electrostatics Maxwell equations
	\2 $\nabla \cdot \mathbf{D} = \rho_{\mathrm{v}}$
	\2 $\nabla \times \mathbf{E} = 0$
	\1 Magnetostatics Maxwell equations
	\2 $\nabla \cdot \mathbf{B} = 0$
	\2 $\nabla \times \mathbf{H} = \mathbf{J}$
	\1 $\vec{J} = \rho_V \vec{u}$
	\1 $\vec{F} = q^\prime \vec{E}$
	\1 $\vec{D} = \epsilon \vec{E}$
	\1 $\vec{D} = \dfrac{\Psi}{\text{Surface Area}}$
	\1 $dV = \dfrac{dW}{q} = -\mathbf{E} \cdot d\mathbf{l}$
	\1 For any scalar quantity, $A$:
	$dA = \nabla A \cdot d\mathbf{l}$

	The symbol  $\nabla$, the greek letter nabla,  is called the del or gradient operator and is defined as

	$$\nabla = \hat{\mathbf{x}} \dfrac{\partial}{\partial x} + \hat{\mathbf{y}} \dfrac{\partial}{\partial y} + \hat{\mathbf{z}} \dfrac{\partial}{\partial z} \qquad \text{(Cartesian)}$$

	\1 Laplacian cross product example
	\begin{equation*}
		\nabla \times \mathbf{B} =
		\begin{vmatrix}
			\mathbf{\hat{x}}            & \mathbf{\hat{y}}            & \mathbf{\hat{z}}            \\
			\frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\
			B_x                         & B_y                         & B_z
		\end{vmatrix}
		= \mathbf{\hat{x}} \left( \frac{\partial B_z}{\partial y} - \frac{\partial B_y}{\partial z} \right) - \mathbf{\hat{y}} \left( \frac{\partial B_z}{\partial x} - \frac{\partial B_x}{\partial z} \right) + \mathbf{\hat{z}} \left( \frac{\partial B_y}{\partial x} - \frac{\partial B_x}{\partial y} \right)
	\end{equation*}

	\1 $\vec{E} = - \nabla V$
	\1 $\vec{p} = q \cdot \vec{d}$
	\1 $\vec{P} = \epsilon_0 \chi_e \vec{E}$
	\1 $\epsilon_r = 1+\chi_e$
	\1 Electric Susceptibility, $\chi_e = \dfrac{\text{Bound charge density}}{\text{Free charge density}}$
	\1 In an electric field polarized dielectric material, $\vec{D} = \epsilon_0 \vec{E} + \vec{P} = \epsilon_0 \vec{E} +  \epsilon_0 \chi_e \vec{E} = \epsilon_0 (1+\chi_e)E = \epsilon E$
	\2 $\vec{D} = \epsilon_0 \vec{E} + \vec{P}$
	\1 Boundary Conditions. At a system of a sphere of some material (area 1), surrounded by some material (area 2)
	\2 $E_{1t} = E_{2t}$
	\2 $\dfrac{D_{1t}}{\epsilon_0} = \dfrac{D_{2t}}{\epsilon_0}$
	\2 $D_{1n} - D_{2n} = \rho_S$
	\1 $V_{br} = E_{ds}d$
	\1 $E = \dfrac{Q}{\epsilon A}$
	\1 $C = \dfrac{Q}{V} = \dfrac{Q}{Ed} = \dfrac{\epsilon A}{d}$
\end{outline}

In a perfect conductor that has an applied electric field, inside the conductor $E_{\text{net}} = 0$
